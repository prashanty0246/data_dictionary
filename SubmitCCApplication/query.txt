const getValidFDList = `select
  Case When (
    Select
      Count(1)
    From
      Tbaadm.Alt
    Where
      Acid = Gam.Acid
      And Entity_Cre_Flg = 'Y'
      And Del_Flg = 'N'
  )> 1 Then (
    Select
      B2k_Type
    From
      Tbaadm.Alt
    Where
      Acid = Gam.Acid
      And B2k_Type != 'SHFLN'
      and ENTITY_CRE_FLG = 'Y'
      and DEL_FLG = 'N'
      And Rownum = 1
  ) Else NVL(
    (
      Select
        B2k_Type
      From
        Tbaadm.Alt
      Where
        Acid = Gam.Acid
        And Entity_Cre_Flg = 'Y'
        And Del_Flg = 'N'
        And Rownum = 1
    ),
    'NA'
  ) end || '|' || --"Lien Code"
  gam.lien_amt || '|' || --"Lien Amt."
  decode(
    (
      select
        deposit_type
      from
        tbaadm.tam
      where
        acid = gam.acid
    ),
    'F',
    'Y',
    'N'
  )|| '|' || --"Is FFD"
  nvl(
    (
      select
        CUSTOMERMINOR
      from
        crmuser.accounts
      where
        orgkey = gam.cif_id

          select

        acid = gam.acid
